Select d.name as Department, t.name as Employee, t.salary as Salary
from (
    select
    e.*, DENSE_Rank()
    over (partition by e.departmentId order by salary desc) as rnk
    From Employee e
) t
JOIN Department d 
on t.departmentId = d.id
where t.rnk <=3;
